### GraphRAG = Perceiving the Pattern of Reflections

```python
class IndrasNetGraphRAG:
    """
    GraphRAG as Indra's Net perception.

    Each query perceives the pattern of reflections across
    the net, retrieving jewels based on their reflection
    relationships and mutual identities.
    """

    def query_by_reflection(
        self,
        query_text: str,
        coordinate_filter: Optional[Coordinate] = None
    ) -> List[JewelReflection]:
        """
        Query jewels by their reflection content.

        In Indra's Net, each jewel reflects all other jewels.
        Query by finding jewels whose reflections match the query.
        """
        # Find jewels that reflect this query
        reflecting_jewels = self.find_reflecting_jewels(
            query_text,
            coordinate_filter
        )

        # Return jewels with their reflection chains
        return [
            JewelReflection(
                jewel=jewel,
                reflects=self.get_all_reflections(jewel),
                is_reflected_in=self.get_mutual_reflections(jewel),
                contains=self.extract_net_presence(jewel)
            )
            for jewel in reflecting_jewels
        ]

    def traverse_reflection_chain(
        self,
        start_jewel: NodeID,
        max_depth: int = 5
    ) -> ReflectionChain:
        """
        Traverse the chain of reflections from a jewel.

        Each reflection contains further reflections, ad infinitum.
        This reveals the infinite mutual identity of the net.
        """
        chain = [start_jewel]

        for depth in range(max_depth):
            # Get jewels reflected at this depth
            current = chain[-1]
            reflections = self.get_reflections(current)
            chain.extend(reflections)

        return ReflectionChain(
            chain=chain,
            infinite_mutual_identity=True,
            whole_in_each_link=True
        )

    def get_net_context(
        self,
        jewel: NodeID,
        context_radius: int = 3
    ) -> NetContext:
        """
        Get the net context around a jewel.

        Returns the subgraph of jewels within reflection radius,
        showing how this jewel inter-is with its context.
        """
        # Get all jewels within reflection radius
        context_jewels = self.get_reflections_within_radius(
            jewel,
            context_radius
        )

        return NetContext(
            center_jewel=jewel,
            interbeing_network=context_jewels,
            being_with=self.analyze_being_with(context_jewels),
            # The whole net is present in this context
            whole_in_context=self.detect_whole_in_context(context_jewels)
        )
```

```python
            memory_jewel=memory_jewel,

            # Memory preserves the net at this moment

            recalled_jewels=matches,
```
